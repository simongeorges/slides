Exemple html.tpl.twig

<!DOCTYPE html>
<html lang="{{ language.language }}" dir="{{ language.dir }}">
<head>
  
  {{ head }}
  {{ head_title }}
  {{ styles }}
</head>
<body class="{{ classes }}" {{ attributes }}>
  {{ page_top }}
  {{ page }}
  {{ scripts }}
  {{ page_bottom }}
</body>
</html>
